```js
/**
 * setTimeout 模拟 setInterval
 */
function myInterval(fn, timeout) {
  const timer = {
    stop: false,
  }
  function interval() {
    if (!timer.stop) {
      fn()
      setTimeout(interval, timeout)
    }
  }
  setTimeout(interval, timeout)

  return timer
}

const timer = myInterval(() => console.log(1), 1000)

setTimeout(() => {
  timer.stop = true
}, 5100)
```

```js
/**
 * setInterval 模拟 setTimeout
 */
function myTimeout(fn, timeout) {
  const timer = setInterval(() => {
    fn()
    clearTimeout(timer)
  }, timeout)

  return timer
}
```

```js
/**
 * 提取 url 参数
 */
const getUrlParams = (url) => {
  const reg = /([^?=&]+)=([^&]+)/g
  let res = {}
  url.replace(reg, (_, p1, p2) => {
    res[p1] = p2
  })
  return res
}

let url = 'https://www.junjin.cn?a=1&b=2&c=[1,2]&d'
console.log(getUrlParams(url))
```

```js
/**
 * 十进制转二进制
 */
const bin2ten = (num) => {
  let res = []
  while (num > 0) {
    res.push(num % 2)
    num = parseInt(num / 2)
  }
  return res.reverse().join('')
}

console.log(bin2ten(5))
console.log(bin2ten(4))
```
